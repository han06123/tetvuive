import pygame, sys, random, math
from pygame.locals import *
WINDOWWIDTH = 600
WINDOWHEIGHT = 600
FPS = 60
SIZE = 4.5 # Kch thc vin n n ra
SPEED_CHANGE_SIZE = 0.05 # Tc nh li ca vin n khi n ra
CHANGE_SPEED = 0.07 # Tc chm li ca vin n
RAD = math.pi/180 # i t radian sang
A_FALL = 1.5 # Gia tc ri t do
NUM_BULLET = 50 # S n n ra trong 1 qu pho
SPEED_MIN = 2 # Tc nh nht ca 1 vin n
SPEED_MAX = 4 # Tc ln nht ca mt vin n
TIME_CREAT_FW = 40 # Khong thi gian lin tip gia 2 ln bn
NUM_FIREWORKS_MAX = 3 # S lng pho ln nht bn ln
NUM_FIREWORKS_MIN = 1 # S lng pho nh nht bn ln
SPEED_FLY_UP_MAX = 12 # Tc ln nht ca vin n bay ln (trc khi n)
SPEED_FLY_UP_MIN = 8 # Tc nh nht ca vin n bay ln (trc khi n)
class Dot(): # Nhng chm theo sau ca mi vin n
def __init__(self, x, y, size, color):
self.x = x
self.y = y
self.size = size
self.color = color
def update(self):
# Gim kch thc chm
if self.size > 0:
self.size -= SPEED_CHANGE_SIZE*5
else:
self.size = 0
def draw(self): # V mt chm
if self.size > 0:
pygame.draw.circle(DISPLAYSURF, self.color, (int(self.x), int(self.y)),
int(self.size))
class BulletFlyUp(): # Vin n bay ln trc khi n
def __init__(self, speed, x):
self.speed = speed
self.x = x
self.y = WINDOWHEIGHT
self.dots = [] # Mt list cc chm theo sau
self.size = SIZE/2
self.color = (255, 255, 100)
def update(self):
self.dots.append(Dot(self.x, self.y, self.size, self.color)) # Mi ln n i qua
s c mt chm thm vo
# Xc nh li v tr vin n
self.y -= self.speed
self.speed -= A_FALL*0.1
# update tng chm
for dot in self.dots:
dot.update()
# Xo nhng chm c kch thc <= 0
for dot in self.dots:
if dot.size <= 0:
self.dots.pop(self.dots.index(dot))
def draw(self):
pygame.draw.circle(DISPLAYSURF, self.color, (int(self.x), int(self.y)),
int(self.size)) # V vin n
# V tng chm
for dot in self.dots:
dot.draw()
class Bullet(): # Vin n sau khi n
def __init__(self, x, y, speed, angle, color):
self.x = x
self.y = y
self.speed = speed
self.angle = angle # Gc hp bi vin n v phng ngang
self.size = SIZE
self.color = color
def update(self):
# Xc nh tc theo 2 phng
speedX = self.speed * math.cos(self.angle*RAD)
speedY = self.speed * -math.sin(self.angle*RAD)
# Xc nh li v tr vin n
self.x += speedX
self.y += speedY
self.y += A_FALL
# Gim tc n
if self.size > 0:
self.size -= SPEED_CHANGE_SIZE
else:
self.size = 0
# Gim kch thc n
if self.speed > 0:
self.speed -= CHANGE_SPEED
else:
self.speed = 0
def draw(self): # V 1 vin n
if self.size > 0:
pygame.draw.circle(DISPLAYSURF, self.color, (int(self.x), int(self.y)),
int(self.size))
class FireWork(): # Qu pho hoa
def __init__(self, x, y):
self.x = x
self.y = y
self.dots = [] # List cc chm theo sau mi vin n
def creatBullets(): # To list cc vin n
bullets = []
color = Random.color()
for i in range(NUM_BULLET):
angle = (360/NUM_BULLET)*i
speed = random.uniform(SPEED_MIN, SPEED_MAX)
bullets.append(Bullet(self.x, self.y, speed, angle, color))
return bullets
self.bullets = creatBullets();
def update(self):
for bullet in self.bullets: # update tng vin n
bullet.update()
self.dots.append(Dot(bullet.x, bullet.y, bullet.size, bullet.color))
for dot in self.dots: # update tng chm
dot.update()
# Xo nhng chm c kch thc <= 0
for dot in self.dots:
if dot.size <= 0:
self.dots.pop(self.dots.index(dot))
def draw(self):
for bullet in self.bullets: # V tng vin n
bullet.draw()
for dot in self.dots: # V tng chm
dot.draw()
class Random():
def __init__(self):
pass
def color(): # To mu ngu nhin (mu sng)
color1 = random.randint(0, 255)
color2 = random.randint(0, 255)
if color1 + color2 >= 255:
color3 = random.randint(0, 255)
else:
color3 = random.randint(255 - color1 - color2, 255)
colorList = [color1, color2, color3]
random.shuffle(colorList)
return colorList
def num_fireworks(): # S pho mi ln bn
return random.randint(NUM_FIREWORKS_MIN, NUM_FIREWORKS_MAX)
def randomBulletFlyUp_speed(): # Tc vin n bay ln
speed = random.uniform(SPEED_FLY_UP_MIN, SPEED_FLY_UP_MAX)
return speed
def randomBulletFlyUp_x(): # V tr vin n bay ln
x = random.randint(int(WINDOWWIDTH*0.2), int(WINDOWHEIGHT*0.8))
return x
def main():
global FPSCLOCK, DISPLAYSURF
pygame.init()
pygame.display.set_caption('FIREWORKS')
FPSCLOCK = pygame.time.Clock()
DISPLAYSURF = pygame.display.set_mode((WINDOWWIDTH, WINDOWHEIGHT))
fireWorks = []
time = TIME_CREAT_FW
bulletFlyUps = []
while True:
DISPLAYSURF.fill((0, 0, 0)) # Xo nn
for event in pygame.event.get():
if event.type == QUIT or (event.type == KEYUP and event.key ==
K_ESCAPE):
pygame.quit()
sys.exit()
if time == TIME_CREAT_FW: # To (nhng) vin n bay ln sau khong
thi gian xc inh
for i in range(Random.num_fireworks()):
bulletFlyUps.append(BulletFlyUp(Random.randomBulletFlyUp_speed(),
Random.randomBulletFlyUp_x()))
for bulletFlyUp in bulletFlyUps:
bulletFlyUp.draw()
bulletFlyUp.update()
for fireWork in fireWorks:
fireWork.draw()
fireWork.update()
for bulletFlyUp in bulletFlyUps:
if bulletFlyUp.speed <= 0: # Vin n bay ln t cao ti a
fireWorks.append(FireWork(bulletFlyUp.x, bulletFlyUp.y)) #
To qu pho ngay v tr vin n
bulletFlyUps.pop(bulletFlyUps.index(bulletFlyUp)) # Xo vin
n
# Xo qu pho hoa khi kch thc nhng vin n <= 0
for fireWork in fireWorks:
if fireWork.bullets[0].size <= 0:
fireWorks.pop(fireWorks.index(fireWork))
# m khong thi gian bn
if time <= TIME_CREAT_FW:
time += 1
else:
time = 0
pygame.display.update()
FPSCLOCK.tick(FPS)
if __name__ == '__main__':
main()
